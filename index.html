// SUBSTITUA pela sua URL real do Render (sem a barra / no final e sem :5000) const API_BASE_URL = "https://seu-app-no-render.onrender.com"; async function enviarAnalise() { const prompt = document.getElementById('userPrompt').value; const responseContainer = document.getElementById('response'); // Mostrar carregamento responseContainer.innerHTML = "A analisar com IA SambaNova... Por favor aguarde."; try { const response = await fetch(`${API_BASE_URL}/api/analyze`, { method: 'POST', headers: { 'Content-Type': 'application/json', }, body: JSON.stringify({ prompt: prompt }) }); if (!response.ok) { throw new Error(`Erro no Servidor: ${response.status}`); } const data = await response.json(); // Exibir a resposta baseada no formato do backend que criámos if (data.agents && data.agents.length > 0) { responseContainer.innerHTML = `

${data.agents[0].area}

${data.agents[0].analysis}

`; } } catch (error) { responseContainer.innerHTML = `Erro de ligação: ${error.message}. Verifique se o backend no Render está 'Live'.`; console.error("Erro detalhado:", error); } }
